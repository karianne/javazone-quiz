override def toString = {
    val builder = new StringBuilder
    builder.append("Q: ")
    builder.append(text)
    builder.append("\n")
    answers foreach {
      answer =>
        builder.append("\t")
        builder.append(answer._1)
        builder.append(": ")
        builder.append(answer._2)
        builder.append("\n")
    }

    builder.toString
  }
